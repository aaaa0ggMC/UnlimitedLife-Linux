363 inline void LogMsg::build_on_consumer(){
364 if(cfg->disable_extra_information)return;
365 [[maybe_unused]] thread_local static bool inited = [&]{
366 sdate.resize(date_str_resize);
367 return false;
368 }();
369 if(cfg->gen_date){
370 static thread_local time_t old_time = 0;
371 time_t rawtime;
372 time(&rawtime);
373 if(rawtime != old_time){
374 struct tm ptminfo;
375 localtime_r(&rawtime,&ptminfo);
376 sdate.clear();
377 sdate.resize(date_str_resize);
378 sdate.resize(snprintf(sdate.data(),date_str_resize,"%02d-%02d-%02d %02d:%02d:%02d",
379 ptminfo.tm_year + 1900, ptminfo.tm_mon + 1, ptminfo.tm_mday,
380 ptminfo.tm_hour, ptminfo.tm_min, ptminfo.tm_sec));
381 old_time = rawtime;
382 }

386 inline std::string_view LogMsg::gen_composed(){
387 if(cfg->disable_extra_information){body.push_back('\n');return body;}
388 [[maybe_unused]] static thread_local bool inited = [&]{
389 scomposed.clear();
390 scomposed.resize(compose_str_resize);
391 return false;
392 }();
393 // 只要保证按照LogMsg的顺序递归而不是lot，那么这个就是valid的
394 if(generated)return scomposed;
395 size_t beg = 0;
396
397 scomposed.clear();
398 scomposed.resize(scomposed.capacity());
399 if(cfg->gen_date)beg += snprintf(scomposed.data() + beg,scomposed.size() - beg,
400 "[%s]",sdate.c_str());
401 if(cfg->out_level)beg += snprintf(scomposed.data() + beg,scomposed.size() - beg,
402 "[%s]",cfg->level_cast(level).data());
403 if(cfg->out_header && header.data() != nullptr && header.size())beg += snprintf(scomposed.data() + beg,scomposed.size() - beg,
404 "[%s]",header.data());
405 if(cfg->gen_time)beg += snprintf(scomposed.data() + beg,scomposed.size() - beg,
406 "[%.2lfms]",timestamp);
407 if(cfg->gen_thread_id)beg += snprintf(scomposed.data() + beg,scomposed.size() - beg,
408 "[TID%lu]",thread_id);
409 scomposed.resize(beg);
410 scomposed.append(":");
411
412 scomposed += body;
413 scomposed.append("\n");
414 generated = true;
415
416 return scomposed;

438 template<CanAccessItem T> inline bool Logger::safe_remove_mod(
439 std::string_view key,
440 std::unordered_map<std::string,RefWrapper<T>> & st,
441 T & container
442 ){
443 auto it = st.find(std::string(key));
444 if(it == st.end()){
445 return false;
446 }
447 // 旧的target不需要手动关闭，因为析构函数自动关闭了
448 // 不懂为什么要先erase it，但是Gemini大王发话了还特别固执
449 size_t cached_index = it->second.index;
450 st.erase(it);
451 // 后面的index都要减少一个
452 for(auto& [key,val] : st){
453 if(val.index > cached_index)val--;
454 }
455 container.erase(container.begin() + cached_index);
456 return true;
457 }

489 template<class T,class... Args> inline std::shared_ptr<T> Logger::append_mod(std::string_view name,Args&&... args){
490 static_assert(std::is_base_of_v<LogTarget,T> || std::is_base_of_v<LogFilter,T>,
491 "T must be the derived class of LogTarget or LogFilter!");
492 auto ptr = std::make_shared<T>(std::forward<Args>(args)...);
493
494 if constexpr(std::is_base_of_v<LogTarget,T>){
496 auto it = search_targets.find(std::string(name));
497 if(it == search_targets.end()){
498 targets.push_back(ptr);
499 search_targets.emplace(name,ref(targets,targets.size() - 1));
500 }else{
501 it->second = ptr;
502 }
503 }else{
504 auto it = search_filters.find(std::string(name));
505 if(it == search_filters.end()){
506 filters.push_back(ptr);
507 search_filters.emplace(name,ref(filters,filters.size() - 1));
508 }else{
509 it->second = ptr;
510 }
511 }
512 return ptr;
513 }

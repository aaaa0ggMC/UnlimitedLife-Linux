178 inline BenchmarkResults run(uint32_t times,uint32_t repeat){
179 BenchmarkResults rs;
180
182 {
183 Clock clk;
184 timespec st;
185 clk.start();
186 uint64_t warmTimes = 0;
187 uint64_t sink = 0;
188 clock_gettime(ALIB_PERF_CLOCK_SOURCE,&st);
189 while(clk.getAllTime() < warm_up_time_ms){
190 for(int i = 0; i < warm_up_per_run_times; ++i)sink += 1;
191 warmTimes += warm_up_per_run_times;
192 sink = 0;
193 }
194 }
195
196 for(uint32_t i = 0;i < repeat;++i){
197 rs.results.push_back(single_run(times));
198 }
199
200 return rs;
201 }

207 inline SingleBenchmarkResult single_run(uint32_t times){
208 SingleBenchmarkResult result = {0};
209 timespec st,ed;
210 clock_gettime(ALIB_PERF_CLOCK_SOURCE,&st);
211 for(uint32_t i = 0;i < times;++i){
212 func();
213 }
214 clock_gettime(ALIB_PERF_CLOCK_SOURCE,&ed);
215 result.timeSum = (ed.tv_sec - st.tv_sec) * 1000 + (ed.tv_nsec - st.tv_nsec) / 1'000'000.0;
216 result.times = times;
217 return result;
218 }
219 #pragma GCC pop_options

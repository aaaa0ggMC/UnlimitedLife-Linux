191 template<CanDump Dumper> inline void dump(Dumper & appender,size_t indent = 4){
192 using namespace detail;
193 std::string indent_str(indent,' ');
194 std::stack<Node*> tree;
195 int depth = 0;
196 if(this->name.empty()){ // 说明是匿名节点，一般一定为root，一定无value，进行一层遍历
197 for(auto & node : this->children){
198 tree.push(&node);
199 }
200 }else tree.push(this);
201
202 static auto add_str = [](Dumper & dmp,std::string_view sv){
203 dump_append(dmp,"\"");
204 for(auto ch : sv){
205 if(ch == '\n'){
206 dump_append(dmp,"\\n");
207 }else if(ch == '\t'){
208 dump_append(dmp,"\\t");
209 }else if(ch == '\\'){
210 dump_append(dmp,"\\\\");
211 }else if(ch == '\"'){
212 dump_append(dmp,"\\\"");
213 }else if(ch =='\''){
214 dump_append(dmp,"\\\'");
215 }else dump_append(dmp,std::string_view(&ch,1));
216 }
217 dump_append(dmp,"\"");
218 };
219
220 while(!tree.empty()){
221 Node * top = tree.top();
222
224 for(int i = 0;i < depth;++i){
225 dump_append(appender,indent_str);
226 }
227
228 add_str(appender,top->name);
229
230 for(auto & value : top->values){
231 dump_append(appender," ");
232 add_str(appender,value);
233 }
234 dump_append(appender," ");
235
236 // Add children
237 if(top->children.size()){
238 dump_append(appender,"{\n");
239 for(auto & node : top->children)tree.push(&node);
240 ++depth;
241 continue; // 说明目前还是父节点
242 }
243
244 tree.pop(); // 目前他没有孩子了
245 if(tree.empty()){
246 dump_append(appender,";\n");
247 break;
248 }
249 Node & new_top = *tree.top();
250 if(new_top.children.size()){
252 int off = top - &(new_top.children[0]);
254 if(off < 0 && off >= sizeof(Node) * new_top.children.size()){
256 dump_append(appender,";\n");
257 }else{
259 tree.pop();
260 --depth;
261 dump_append(appender,";\n");
262 for(int i = 0;i < depth;++i){
263 dump_append(appender,indent_str);
264 }
265 dump_append(appender,"};\n");
266 }
267 }else{
268 //上一个元素没有子节点，说明应该是平级
269 dump_append(appender,";\n");
270 }
271 }
272 }

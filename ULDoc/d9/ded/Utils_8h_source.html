294 class EventLoop {
295 public:
296 alib::g3::Clock clock;
297 alib::g3::RateLimiter ratelimiter = alib::g3::RateLimiter(1000.0f);
298 std::vector<Event> events;
299 bool isRunning;
300
301 inline void setInterval(const std::function<void()>&task, int64_t time){
302 int64_t now = (int64_t) clock.getAllTime();
303 Event event;
304 event.startOn = now + time;
305 event.task = task;
306 event.interval = time;
307 addEvent(event);
308 }
309
310 inline void setTimeout(const std::function<void()>& task, int64_t time){
311 int64_t now = (int64_t) clock.getAllTime();
312 Event event;
313 event.startOn = now + time;
314 event.task = task;
315 addEvent(event);
316 }
317
318 inline void addEvent(Event& event){
319 events.push_back(event);
320 }
321
322 inline void loop(){
323 while (isRunning){
324 std::vector<int> toErase;
325 // 事件循环
326 int size = events.size();
327 for (int index = 0;index < size;index++){
328 auto& event = events[index];
329 int64_t now = (int64_t) clock.getAllTime();
330 if (now >= event.startOn){
331 event.task();
332 if (!isRunning) break;
333 if (event.interval > 0){
334 event.startOn = now + event.interval;
335 }else {
336 toErase.push_back(index);
337 }
338 }
339 }
340 // 删除事件
341 for (int& i : toErase){
342 events.erase(events.begin() + i);
343 }
344 ratelimiter.wait();
345 }
346 }
347
348 inline void start(){
349 isRunning = true;
350 loop();
351 }
352
353 inline void stop(){
354 isRunning = false;
355 }
356 };

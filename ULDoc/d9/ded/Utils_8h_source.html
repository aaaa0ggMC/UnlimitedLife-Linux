248 class EventLoop {
249 public:

252 std::vector<Event> events;
253 bool isRunning;

255 inline void setInterval(const std::function<void()>&task, int64_t time){
256 int64_t now = (int64_t) clock.getAllTime();
257 Event event;
258 event.startOn = now + time;
259 event.task = task;
260 event.interval = time;
261 addEvent(event);
262 }
263
264 inline void setTimeout(const std::function<void()>& task, int64_t time){
265 int64_t now = (int64_t) clock.getAllTime();
266 Event event;
267 event.startOn = now + time;
268 event.task = task;
269 addEvent(event);
270 }
271
272 inline void addEvent(Event& event){
273 events.push_back(event);
274 }
275
276 inline void loop(){
277 while (isRunning){
278 std::vector<int> toErase;
279 // 事件循环
280 int size = events.size();
281 for (int index = 0;index < size;index++){
282 auto& event = events[index];
283 int64_t now = (int64_t) clock.getAllTime();
284 if (now >= event.startOn){
285 event.task();
286 if (!isRunning) break;
287 if (event.interval > 0){
288 event.startOn = now + event.interval;
289 }else {
290 toErase.push_back(index);
291 }
292 }
293 }
294 // 删除事件
295 for (int& i : toErase){
296 events.erase(events.begin() + i);
297 }
298 ratelimiter.wait();
299 }
300 }
301
302 inline void start(){
303 isRunning = true;
304 loop();
305 }
306
307 inline void stop(){
308 isRunning = false;
309 }
310 };

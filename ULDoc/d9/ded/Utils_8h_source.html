303 inline void setInterval(const std::function<void()>&task, int64_t time){
304 int64_t now = (int64_t) clock.getAllTime();
305 Event event;
306 event.startOn = now + time;
307 event.task = task;
308 event.interval = time;
309 addEvent(event);

312 inline void setTimeout(const std::function<void()>& task, int64_t time){
313 int64_t now = (int64_t) clock.getAllTime();
314 Event event;
315 event.startOn = now + time;
316 event.task = task;
317 addEvent(event);
318 }
319
320 inline void addEvent(Event& event){
321 events.push_back(event);
322 }
323
324 inline void loop(){
325 while (isRunning){
326 std::vector<int> toErase;
327 // 事件循环
328 int size = events.size();
329 for (int index = 0;index < size;index++){
330 auto& event = events[index];
331 int64_t now = (int64_t) clock.getAllTime();
332 if (now >= event.startOn){
333 event.task();
334 if (!isRunning) break;
335 if (event.interval > 0){
336 event.startOn = now + event.interval;
337 }else {
338 toErase.push_back(index);
339 }
340 }
341 }
342 // 删除事件
343 for (int& i : toErase){
344 events.erase(events.begin() + i);
345 }
346 ratelimiter.wait();
347 }
348 }
349
350 inline void start(){
351 isRunning = true;
352 loop();
353 }
354
355 inline void stop(){
356 isRunning = false;
357 }

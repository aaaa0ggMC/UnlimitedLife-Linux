131 struct DLL_EXPORT RotateFile : public LogTarget{
132 private:
133 FILE * f {nullptr};
134 int rotate_index { 0 };
135 bool noneed_update {false};
136 uint64_t bytes_written {0};
137 std::string current_fp_delayed;
138 RotateFileConfig config;
139 timespec last_expire {-1,-1};
140
141 inline bool expires(){
142 timespec tm;
143 clock_gettime(time_clock_source,&tm);
144 // 大小超了
145 bool opt = config.rotate_size && (bytes_written >= config.rotate_size);
146 defer{
147 if(opt){
148 last_expire = tm;
149 }
150 };
151 if(opt)return true;
152 // 时间超了
153 if(last_expire.tv_nsec < 0){
154 // 尚未初始化
155 last_expire = tm;
156 opt = false;
157 }else opt = config.rotate_time > 0 && (((tm.tv_sec - last_expire.tv_sec) * 1000 + (tm.tv_nsec - last_expire.tv_nsec) / 1'000'000)
158 >= config.rotate_time);
159 if(opt)return true;
160 return false;
161 }
162
163 public:
164 inline std::string_view get_current_filepath(){
165 if(noneed_update)return current_fp_delayed;
166 noneed_update = true;
167 current_fp_delayed.clear();
168
169 time_t curtime = time(0);
170
171 std::vformat_to(std::back_inserter(current_fp_delayed),
172 config.filepath_fmt,std::make_format_args(curtime,rotate_index));
173 return current_fp_delayed;
174 }
175
176 inline void flush() override{
177 if(f)fflush(f);
178 }
179
180 inline void try_open(){
181 if(f && !expires())return;
182 close();
183 noneed_update = false;
184 f = fopen(get_current_filepath().data(),"w");
185 // 重置写入量
186 panicf_debug(!f,"Cannot open file {}!",get_current_filepath());
187 if(!f && config.failed_open_fn){
188 config.failed_open_fn(get_current_filepath(),*this);
189 }else{
190 ++rotate_index;
191 }
192 }

196 fclose(f);
197 bytes_written = 0;
198 f = nullptr;
199 }
200 }

207 }

244 // 在flush中刷新

246 if(fn){
247 auto ptr = fn(msg);
248 if(!ptr.empty()){
249 __internal_alib_fw(ptr.data(),sizeof(decltype(ptr)::value_type),ptr.size(),stdout);
250 return true;
251 }
252 }
253 return false;
254 };
255 static auto reset_color = [](){
256 __internal_alib_fw(c_reset_color.data(),sizeof(decltype(c_reset_color)::value_type),c_reset_color.size(),stdout);
257 };
258 {

260 if(!msg.cfg.disable_extra_information){

262 __internal_alib_fw("[",1,1,stdout);

264 __internal_alib_fw(msg.sdate.c_str(),sizeof(decltype(msg.sdate)::value_type),msg.sdate.size(),stdout);
265 if(val)reset_color();

267 }

272 __internal_alib_fw(ls.data(),sizeof(decltype(ls)::value_type),ls.size(),stdout);
273 if(val)reset_color();
274 __internal_alib_fw("]",1,1,stdout);
275 }

277 __internal_alib_fw("[",1,1,stdout);

280 if(val)reset_color();

283 if(msg.cfg.gen_time){
284 std::string_view use_color;

286 printf("[%s%.2lfms%s]",use_color.data(),msg.timestamp,c_reset_color.data());

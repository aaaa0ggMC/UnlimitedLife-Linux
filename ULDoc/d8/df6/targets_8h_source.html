129 inline bool expires(){
130 timespec tm;
131 clock_gettime(time_clock_source,&tm);
132 // 大小超了
133 bool opt = config.rotate_size && (bytes_written >= config.rotate_size);
134 defer{
135 if(opt){
136 last_expire = tm;
137 }
138 };
139 if(opt)return true;
140 // 时间超了
141 if(last_expire.tv_nsec < 0){
142 // 尚未初始化
143 last_expire = tm;
144 opt = false;
145 }else opt = config.rotate_time > 0 && (((tm.tv_sec - last_expire.tv_sec) * 1000 + (tm.tv_nsec - last_expire.tv_nsec) / 1'000'000)
146 >= config.rotate_time);
147 if(opt)return true;
148 return false;
149 }

151 public:
152 inline std::string_view get_current_filepath(){
153 if(noneed_update)return current_fp_delayed;
154 noneed_update = true;
155 current_fp_delayed.clear();
156
157 time_t curtime = time(0);
158
159 std::vformat_to(std::back_inserter(current_fp_delayed),
160 config.filepath_fmt,std::make_format_args(curtime,rotate_index));

209 inline std::string_view ConsoleConfig::default_level_color_schema(const LogMsg & msg){
210 switch(msg.level){
211 case 0: // Trace采用白色
212 return "\e[37m";
213 case 1: // Debug采用紫色
214 return "\e[35m";
215 case 2: // Info采用黄色
216 return "\e[33m";
217 case 3: // Warn采用蓝色
218 return "\e[34m";
219 case 4: // Error采用红色
220 return "\e[31m";
221 case 5: // Fatal白底红字
222 return "\e[31;47m";
223 default:
224 return "";

228 inline void Console::write(
229 LogMsg & msg
230 ){
231 static std::string_view c_reset_color = "\e[0m"; // 复原颜色
232 // 在flush中刷新
233 static auto fast_print = [](ConsoleConfig::ColorSchemaFn fn,LogMsg & msg){
234 if(fn){
235 auto ptr = fn(msg);
236 if(!ptr.empty()){
237 fwrite_unlocked(ptr.data(),sizeof(decltype(ptr)::value_type),ptr.size(),stdout);
238 return true;
239 }
240 }
241 return false;
242 };
243 static auto reset_color = [](){
244 fwrite_unlocked(c_reset_color.data(),sizeof(decltype(c_reset_color)::value_type),c_reset_color.size(),stdout);
245 };
246 {
247 std::lock_guard<std::mutex> lock(console_lock);
248 if(!msg.cfg->disable_extra_information){
249 if(msg.cfg->gen_date){
250 fwrite_unlocked("[",1,1,stdout);
251 bool val = fast_print(cfg.date_color_schema,msg);
252 fwrite_unlocked(msg.sdate.c_str(),sizeof(decltype(msg.sdate)::value_type),msg.sdate.size(),stdout);
253 if(val)reset_color();
254 fwrite_unlocked("]",1,1,stdout);
255 }
256 if(msg.cfg->out_level){
257 fwrite_unlocked("[",1,1,stdout);
258 bool val = fast_print(cfg.level_color_schema,msg);
259 auto ls = msg.cfg->level_cast(msg.level);
260 fwrite_unlocked(ls.data(),sizeof(decltype(ls)::value_type),ls.size(),stdout);
261 if(val)reset_color();
262 fwrite_unlocked("]",1,1,stdout);
263 }
264 if(msg.cfg->out_header && msg.header.data() != nullptr && msg.header.size()){
265 fwrite_unlocked("[",1,1,stdout);
266 bool val = fast_print(cfg.head_color_schema,msg);
267 fwrite_unlocked(msg.header.data(),sizeof(decltype(msg.header)::value_type),msg.header.size(),stdout);
268 if(val)reset_color();
269 fwrite_unlocked("]",1,1,stdout);
270 }
271 if(msg.cfg->gen_time){
272 std::string_view use_color;
273 if(cfg.time_color_schema && !(use_color = cfg.time_color_schema(msg)).empty()){
274 printf("[%s%.2lfms%s]",use_color.data(),msg.timestamp,c_reset_color.data());
275 }else{
276 printf("[%.2lfms]",msg.timestamp);
277 }
278 }
279 if(msg.cfg->gen_thread_id){
280 std::string_view use_color;
281 if(cfg.thread_id_color_schema && !(use_color = cfg.thread_id_color_schema(msg)).empty()){
282 printf("[%sTID%lu%s]",use_color.data(),msg.thread_id,c_reset_color.data());
283 }else{
284 printf("[TID%lu]",msg.thread_id);
285 }
286 }
287 putchar_unlocked(':');
288 }
289 fwrite_unlocked(msg.body.data(),sizeof(decltype(msg.body)::value_type),msg.body.size(),stdout);
290 putchar_unlocked('\n');
291 }
292 }
